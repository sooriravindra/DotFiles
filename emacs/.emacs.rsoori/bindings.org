#+begin_src emacs-lisp :results value file :file bindings.el :var table=BindingsTable
  (let ((evilmap "")
        (wkmap "")
        (wkadd "which-key-add-key-based-replacements"))
    (progn
      (cl-loop for row in (cdr table)
               for desc = (nth 2 row)
               for func = (nth 1 row)
               for keys = (car row)
               do (if (eq func "")
                      (setq wkmap
                            (concat wkmap
                                    (format "\n(%s \"SPC %s\" \"%s\")"
                                            wkadd keys desc)
                                    (format "\n(%s \"C-SPC %s\" \"%s\")"
                                            wkadd keys desc)))
                    (setq evilmap
                          (concat evilmap
                                  (format "\n(kbd \"<leader> %s\") '(\"%s\" . %s)"
                                          keys desc func)))))
      (concat (format "(evil-define-key nil 'global %s )\n" evilmap)
              wkmap)))
#+end_src
